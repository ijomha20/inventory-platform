    $(document).ready(function () {
        // Check with the server if the FTUJ has been completed
        $.ajax({
            type: 'GET',
            url: '/api/ajax/GetCompletedActionByUserId?completedAction=AutoFraudFTUJ',
            success: function (data) {
                // If the server returns false, that means the FTUJ hasn't been completed
                if (data === false) {
                    $('#ftujModal').modal('show');
                } else {
                    $('#ftujModal').modal('hide');
                }
            },
            error: function () {
                console.error("Error checking if FTUJ has been completed.");
            }
        });
        // Handle the "Got It" button click event
        $('.gotIt-button').click(function () {
            // Call the completeAF function to mark the FTUJ as completed
            completeAF();
            $('#ftujModal').modal('close');
        });
        // Handle the "What we check for" button click event
        $('.check-button').click(function () {
            $('#ftujModal').modal('hide');
            whatWeCheckFor();
            completeAF();
            $(document).ready(function () {
                $('#ftujModal').modal('close'); // Ensure modal stays closed on page load
            });

        });
    });
    // Function to send the AJAX request to mark AF FTUJ as completed
    function completeAF() {
        var input = '{ "UserId": "85385", "CompletedAction" : "AutoFraudFTUJ" }';

        $.ajax({
           type: 'POST',
           url: '/api/ajax/SetCompletedAction',
           contentType: 'application/json',
           dataType: 'json',
           data: input,
    });
}

    function whatWeCheckFor() {
        var whatWeCheckForUrl = "/Products/VINFraudCheck";
        window.open(whatWeCheckForUrl, "_blank");
    }

            let activeStolenFlags = [isStolen];

                function convertReportType(reportType) {
                    switch (reportType.toLowerCase()) {
                        case ReportTypes.ARAA:
                            return 'ARAA';
                        case ReportTypes.Verified:
                            return 'Vehicle History Report + Lien Check';
                        case ReportTypes.VerifiedBC:
                            return 'Vehicle History Report + Lien Check (BC)';
                        case ReportTypes.ClaimsBC:
                            return 'Vehicle History Report (BC)';
                        case ReportTypes.Claims:
                            return 'Vehicle History Report';
                        case ReportTypes.Lien:
                            return 'CARFAX Canada Lien';
                        case ReportTypes.LienExpress:
                            return 'CARFAX Canada Lien Express';
                        case ReportTypes.USExpress:
                            return 'US Express';
                        case ReportTypes.AlbertaUVIP:
                            return 'Alberta UVIP';
                        case ReportTypes.QuickProof:
                            return 'QuickProof';
                        case ReportTypes.ODVR:
                            return reportType;
                        default:
                            return reportType;
                    }
                }

                function viewCpo(element) {
                    var w = 475;
                    var h = 310;

                    var type = $(element).data('cpo-type');
                    var id = $(element).data('cpo-id');

                    // type vw en /vw_cpo.aspx?id=
                    // type vw fr /fr/cpo/vwcpo?id=
                    // type audi en /audi_cpo.aspx?id=
                    // type audi fr /fr/cpo/audicpo?id=

                    var fullUrl = 'https://vhr-legacy.carfax.ca';

                    if (type === 'vw') {
                        fullUrl = fullUrl + (thisInstance.language !== 'fr-CA' ? '/vw_cpo.aspx?id=' : '/fr/cpo/vwcpo?id=');
                    } else {
                        fullUrl = fullUrl + (thisInstance.language !== 'fr-CA' ? '/audi_cpo.aspx?id=' : '/fr/cpo/audicpo?id=');
                    }

                    fullUrl = fullUrl + id;

                    if (fullUrl.length > 0) {
                        if (w === 0 || w === null)
